3. 資料庫邏輯實作 (SQL & Backend Logic)
為了支持這個計算，我們需要透過 SQL 將 Contribution_Logs（紀錄流量）與 Enterprises（紀錄匯率）進行關聯計算。

SQL
-- 模擬「以課易課」的清算邏輯
SELECT 
    a.title AS course_name,
    a.contribution_pts_reward AS base_pts, -- 基礎上傳獎勵
    COUNT(l.log_id) * 10 AS view_pts, -- 假設每 1 次分享點閱給 10 Pts
    e.vip_level,
    -- 根據 VIP 等級給予額外加權 (如 Lv.2 額外給予 5% 獎勵)
    CASE 
        WHEN e.vip_level >= 2 THEN (COUNT(l.log_id) * 10) * 0.05 
        ELSE 0 
    END AS vip_bonus
FROM Assets a
JOIN Contribution_Logs l ON a.asset_id = l.asset_id
JOIN Users u ON l.user_id = u.user_id
JOIN Enterprises e ON u.enterprise_id = e.enterprise_id
WHERE a.asset_id = 'course-uuid-here' 
  AND l.contribution_type = '分享點閱'
GROUP BY a.asset_id, a.title, a.contribution_pts_reward, e.vip_level;

1. 企業功能開關與權限過濾 API
這個 API 用於前端載入時，決定顯示哪些素材與功能。

JavaScript
const express = require('express');
const router = express.Router();
const db = require('./db'); // 假設的 DB 連線

// 取得過濾後的資產清單 (根據角色與場景)
router.get('/assets/filtered-list', async (req, res) => {
    const { userId, context } = req.query; // context: 'DIGITAL_CARD' 或 'BRAND_VIDEO'

    try {
        // 1. 取得使用者角色與企業 ID
        const user = await db('Users').where({ user_id: userId }).first();
        
        // 2. 檢查企業是否具備該功能的合約權限 (Enterprise_Feature_Status)
        const featureStatus = await db('Enterprise_Feature_Status')
            .where({ 
                enterprise_id: user.enterprise_id, 
                feature_code: context === 'BRAND_VIDEO' ? 'AI_VIDEO' : 'WELFATE_MGT',
                is_enabled: true 
            })
            .where('contract_end_date', '>=', new Date())
            .first();

        if (!featureStatus) {
            return res.status(403).json({ error: '您的企業尚未開通此功能或合約已到期' });
        }

        // 3. 根據角色與場景取得素材策略 (Asset_Usage_Policy)
        const policy = await db('Asset_Usage_Policy')
            .where({ 
                enterprise_id: user.enterprise_id, 
                role_name: user.user_role,
                usage_context: context 
            }).first();

        // 4. 根據策略拉取素材
        const allowedTypes = policy.allowed_asset_types; // 例如 ["IMAGE", "LOGO"]
        const assets = await db('Assets')
            .where({ owner_enterprise_id: user.enterprise_id })
            .whereIn('asset_type', allowedTypes)
            .andWhere('is_compliant', true);

        res.json({ assets, canUseAI: policy.can_use_ai_tools });
    } catch (error) {
        res.status(500).json({ error: '系統錯誤' });
    }
});
2. 福利品下單與物流串接 API
實作點數扣除與 Welfare_Orders 寫入。

JavaScript
router.post('/welfare/redeem-order', async (req, res) => {
    const { userId, assetId, shippingAddress } = req.body;

    const trx = await db.transaction();
    try {
        // 1. 檢查點數是否充足
        const user = await trx('Users').where({ user_id: userId }).forUpdate().first();
        const item = await trx('Assets').where({ asset_id: assetId }).first();

        if (user.wallet_points < item.required_points) {
            throw new Error('點數不足');
        }

        // 2. 扣除點數並記錄交易
        await trx('Users').where({ user_id: userId }).decrement('wallet_points', item.required_points);
        
        const [order] = await trx('Welfare_Orders').insert({
            buyer_user_id: userId,
            asset_id: assetId,
            points_spent: item.required_points,
            order_status: 'PAID',
            shipping_address: shippingAddress
        }).returning('*');

        await trx('Transaction_Ledger').insert({
            user_id: userId,
            tx_type: 'CONSUME',
            category: 'WELFARE',
            amount_pts: item.required_points,
            reference_id: order.order_id
        });

        await trx.commit();
        res.json({ message: '兌換成功', orderId: order.order_id });
    } catch (error) {
        await trx.rollback();
        res.status(400).json({ error: error.message });
    }
});

Gemini AI 自動核銷回傳流程邏輯
非同步觸發： 後端調用 Gemini 生成 API 後，不直接等待結果，而是提供一個 callback_url。

多維度檢核： Gemini 回傳結果時，同步夾帶 AI 檢核報告（LOGO 位置、色碼正確性、合規文字）。

自動結算： 只有檢核通過（例如分數 > 80），系統才會將狀態改為 SUCCESS 並正式完成點數核銷。

一、 回傳處理 API 實作 (Node.js)
此 API 負責接收 AI 運算後的結果並執行資料庫核銷。

JavaScript
const express = require('express');
const router = express.Router();
const db = require('./db');

/**
 * [POST] /api/v1/ai/callback
 * 接收 Gemini 生成任務完成後的通知
 */
router.post('/ai/callback', async (req, res) => {
    const { taskId, outputUrl, auditScore, auditReport, status } = req.body;

    const trx = await db.transaction();
    try {
        // 1. 查找任務並鎖定，防止重複處理 (Idempotency)
        const task = await trx('AI_Generation_Tasks')
            .where({ task_id: taskId, status: 'PROCESSING' })
            .forUpdate()
            .first();

        if (!task) {
            return res.status(200).json({ message: '任務已處理或不存在' });
        }

        if (status === 'COMPLETED' && auditScore >= 80) {
            // 2. 檢核通過：正式更新任務狀態與產出連結
            await trx('AI_Generation_Tasks')
                .where({ task_id: taskId })
                .update({
                    output_url: outputUrl,
                    vi_audit_score: auditScore,
                    status: 'SUCCESS',
                    completed_at: new Date()
                });

            // 3. 寫入合規日誌 (對應 Compliance_Audit_Logs 表)
            await trx('Compliance_Audit_Logs').insert({
                asset_id: task.source_asset_id,
                audit_type: 'VI_CHECK',
                score: auditScore,
                is_passed: true,
                ai_feedback_report: JSON.stringify(auditReport)
            });

            // 4. 正式寫入 Transaction_Ledger (此時點數才算真正「消耗」完畢)
            await trx('Transaction_Ledger').insert({
                user_id: task.user_id,
                tx_type: 'CONSUME',
                category: 'AI_SERVICE',
                amount_pts: task.points_reserved,
                reference_id: taskId,
                description: `AI 生成成功核銷: 任務 ${taskId}`
            });

        } else {
            // 5. 檢核失敗：標記失敗原因，並退還預扣點數 (選配)
            await trx('AI_Generation_Tasks')
                .where({ task_id: taskId })
                .update({
                    status: 'FAILED',
                    vi_audit_score: auditScore,
                    error_log: '品牌檢核未通過'
                });
            
            // 退還點數邏輯...
        }

        await trx.commit();
        res.status(200).send('CALLBACK_PROCESSED');

    } catch (error) {
        await trx.rollback();
        console.error('AI Callback 處理失敗:', error);
        res.status(500).send('RETRY_LATER');
    }
});
