CREATE TABLE Enterprise_Feature_Status (

    enterprise_id UUID REFERENCES Enterprises(enterprise_id) ON DELETE CASCADE,

    -- 合約生效與到期日 (核心管理功能)
    contract_start_date DATE,
    contract_end_date DATE,
    
    -- 備註 (紀錄合約變更或專案特案需求)
    remarks TEXT,
    
    updated_at TIMESTAMP DEFAULT NOW(),
    
    -- 確保同一企業下，同一個功能代碼不會重複出現
    CONSTRAINT unique_enterprise_feature UNIQUE(enterprise_id, feature_code)
);
